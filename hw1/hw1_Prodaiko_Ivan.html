df <- read.csv(file="./GLB.Ts+dSST.csv", header=TRUE, sep=",")
df$Year <- as.Date(as.yearmon(df$Year))
df$annual_mean <- apply(df[, 2:13], 1, mean)
baseline <- mean(df$annual_mean[1:20])
df$adjusted_mean_nyt <- df$annual_mean + abs(baseline)

labels <- c(1904, 1944, 1998, 2014, 2015, 2016, 2017)

dgr_fmt <- function(x) {
  parse(text = sprintf("%+7.2f^o", x))
}

ggplot(df, aes(Year, adjusted_mean_nyt)) +
  geom_hline(yintercept = 0, colour="#000000", alpha = 0.4) +
  geom_text(
    aes(as.Date(as.yearmon(2010)), 0.1), 
    label="HOTTER THAN THE\n1880-1899 AVERAGE",
    size=2
  ) +
  geom_text(
    aes(as.Date(as.yearmon(2010)), -0.05), 
    label="COLDER",
    size=2,
    hjust=-.2
  ) +
  geom_point(shape = 21, colour = "black", fill = "#e4711f", size = 2, stroke = 0.2) +
  geom_text(
    aes(
      label=ifelse(as.numeric(format(df$Year, "%Y")) %in% labels, format(df$Year, "%Y"), NA),
      hjust=1.5,
      vjust=.5
    ), 
    size=3, 
    na.rm=T
  ) +
  scale_x_date(
    date_breaks = "10 years",
    date_labels = "%Y",
    limits = c(min(df$Year), max = max(df$Year) + 365 * 2.5),
    expand=c(0, 0)
  ) +
  scale_y_continuous(
    labels = dgr_fmt, 
    limits = c(-.4, 1.3), 
    breaks = seq(from = -.4, to=1.3, by = 0.2)
  ) +
  theme_minimal() +
  theme(
    panel.grid.minor = element_blank(),
    panel.grid.major = element_line(colour = "#DADADA",size=0.1),
    plot.title = element_text(face="bold", size=9, hjust=0.05, vjust=-15),
    plot.caption=element_text(colour = "grey", hjust = -0.1),
    axis.text = element_text(colour = "grey", size=8)
  ) +
  labs(
    title="Annual Global Surface Temperature,\nRelative to Late 19th Century Average",
    caption = "Source: NASA",
    x="", y=""
  )

dgr_fmt2 <- function(x) {
  parse(text = sprintf("%+.2f^o~F", x))
}

df2 <- melt(df, id.vars = 'Year', measure.vars = names(df)[2:13])
head(df2)
df2$Year <- as.numeric(format(df2$Year, "%Y"))
minY <- min(df2$Year)
maxY <- max(df2$Year)

minVar <- min(df2$value)
maxVar <- max(df2$value)

ggplot(df2, aes(y = Year)) +
  geom_hline(mapping=aes(yintercept=df2$Year, colour=value), size = 2) +
  facet_grid(~variable) +
  scale_y_continuous(
    trans = 'reverse',
    expand = c(0, 0),
    breaks = c(seq(minY, maxY, by=20), 2017)
  ) +
  scale_colour_gradient2(
    low = "#5e83b6", 
    high = "#e99ca4",
    mid = "white", 
    midpoint = 0,
    labels = dgr_fmt2,
    breaks = c(minVar, 0, maxVar),
    guide = guide_colorbar(
       barwidth = 10, 
       barheight = 0.75,
       label.vjust = 4.5
    ),
    limits=c(min(df2$value), max(df2$value))
  ) +
  theme_minimal() +
  theme(
    panel.grid.minor = element_blank(),
    panel.grid.major = element_line(colour = "#DADADA",size=0.1),
    plot.title = element_text(face="bold", hjust = -0.23),
    plot.subtitle = element_text(hjust = -0.165, vjust=-102),
    plot.caption=element_text(hjust = -0.085),
    axis.text = element_text(size=8),
